## 不可用

<Playground>
{class C extends React.Component {
    constructor() {
        super();
        this.state = {
            disabled: true
        }
    }
    
    
    toggle(){
        this.setState({
            disabled: !this.state.disabled
        })
    }

    render() {
        return (
            <div>
                <Switch disabled={this.state.disabled} defaultChecked />
                <br /><br />
                <Button type="primary" onClick={this.toggle.bind(this)}>Toggle disabled</Button>
            </div>
        );
    }

}}

</Playground>
